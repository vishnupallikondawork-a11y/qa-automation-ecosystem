# 📄 `auth_api.py`

```python id="jlwmdv"
import requests

from utils.config import API_BASE_URL
from utils.logger import get_logger


class AuthAPI:

    def __init__(self):

        self.logger = get_logger()

    def login(self, email, password):

        self.logger.info(
            "Sending login API request"
        )

        payload = {
            "email": email,
            "password": password
        }

        response = requests.post(
            f"{API_BASE_URL}/verifyLogin",
            data=payload
        )

        self.logger.info(
            f"Response Status: "
            f"{response.status_code}"
        )

        return response
```

# 🚀 STORY 4 — API VALIDATION IMPROVEMENTS

---

# 📄 `api_utils.py`

```python id="jlwmo5"
class APIUtils:

    @staticmethod
    def validate_status_code(
            response,
            expected_code
    ):

        assert response.status_code == expected_code, (
            f"Expected {expected_code}, "
            f"got {response.status_code}"
        )

    @staticmethod
    def validate_response_code(
            response_body,
            expected_code
    ):

        assert (
            response_body["responseCode"]
            == expected_code
        )

    @staticmethod
    def validate_key_exists(
            data,
            key
    ):

        assert key in data, (
            f"{key} not found in response"
        )
```
